generic_aerial_bleriotxi_01_diorama_entity = {
	entity = "generic_aerial_bleriotxi_01_entity"

	group = "aerialrecon"

	trigger = {
		OR = {
			scope:military_formation = {
				OR = { 
					has_mobilization_option = mobilization_option:mobilization_option_aerial_recon
					has_mobilization_option = mobilization_option:mobilization_option_cryptography_recon
					has_mobilization_option = mobilization_option:mobilization_option_satellite_recon
				}
			}
			scope:combat_unit_type = unit_type:combat_unit_type_airforce_1
			scope:combat_unit_type = unit_type:combat_unit_type_airforce_2
			scope:combat_unit_type = unit_type:combat_unit_type_airforce_3
