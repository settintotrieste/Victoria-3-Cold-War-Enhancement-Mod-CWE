je_genocide_system = {
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	possible = {
	
		#Genocide law is enacted
		has_law = law_type:law_slavery_cultural
		
	}

	immediate = {
		
	}
	
	invalid = {
	
		OR = {
		
			#Genocide law has been repealed
			NOT = { has_law = law_type:law_slavery_cultural }
			
			#THIS is a democracy
			is_a_democracy = yes
			
		}
		
	}
	
	on_invalid = {
	
		#Reset genocide discovery
		remove_variable = genocide_discovered
		remove_variable = genocide_discovered_reacted
		
		#Remove genocide laws
		if = {
			limit = { is_a_democracy = yes }
			activate_law = law_type:law_slavery_banned
		}
		else = {
			activate_law = law_type:law_slavery_political
		}
		
	}
	
	on_monthly_pulse = {
		
		effect = {
		
			#Kill max 1% of a random discrimated pop every month in the whole country (scaled according to gulag institutions size)
			if = {
				limit = { institution_investment_level = { institution = institution_gulag value = 1 } }
				random_scope_pop = {
					limit = { 
						OR = { 
							culture_accepted = no
							religion_accepted = no
						}
					}
					culture = { save_scope_as = discrimated_culture_to_kill }
					religion = { save_scope_as = discriminated_religion_to_kill }
				}
				kill_population_percent = {
					percent = 0.001
					culture = scope:discrimated_culture_to_kill
				}
				kill_population_percent = {
					percent = 0.001
					religion = scope:discriminated_religion_to_kill
				}
			}
			if = {
				limit = { institution_investment_level = { institution = institution_gulag value = 2 } }
				random_scope_pop = {
					limit = { 
						OR = { 
							culture_accepted = no
							religion_accepted = no
						}
					}
					culture = { save_scope_as = discrimated_culture_to_kill }
					religion = { save_scope_as = discriminated_religion_to_kill }
				}
				kill_population_percent = {
					percent = 0.002
					culture = scope:discrimated_culture_to_kill
				}
				kill_population_percent = {
					percent = 0.002
					religion = scope:discriminated_religion_to_kill
				}
			}
			if = {
				limit = { institution_investment_level = { institution = institution_gulag value = 3 } }
				random_scope_pop = {
					limit = { 
						OR = { 
							culture_accepted = no
							religion_accepted = no
						}
					}
					culture = { save_scope_as = discrimated_culture_to_kill }
					religion = { save_scope_as = discriminated_religion_to_kill }
				}
				kill_population_percent = {
					percent = 0.003
					culture = scope:discrimated_culture_to_kill
				}
				kill_population_percent = {
					percent = 0.003
					religion = scope:discriminated_religion_to_kill
				}
			}
			if = {
				limit = { institution_investment_level = { institution = institution_gulag value = 4 } }
				random_scope_pop = {
					limit = { 
						OR = { 
							culture_accepted = no
							religion_accepted = no
						}
					}
					culture = { save_scope_as = discrimated_culture_to_kill }
					religion = { save_scope_as = discriminated_religion_to_kill }
				}
				kill_population_percent = {
					percent = 0.004
					culture = scope:discrimated_culture_to_kill
				}
				kill_population_percent = {
					percent = 0.004
					religion = scope:discriminated_religion_to_kill
				}
			}
			if = {
				limit = { institution_investment_level = { institution = institution_gulag value = 5 } }
				random_scope_pop = {
					limit = { 
						OR = { 
							culture_accepted = no
							religion_accepted = no
						}
					}
					culture = { save_scope_as = discrimated_culture_to_kill }
					religion = { save_scope_as = discriminated_religion_to_kill }
				}
				kill_population_percent = {
					percent = 0.005
					culture = scope:discrimated_culture_to_kill
				}
				kill_population_percent = {
					percent = 0.005
					religion = scope:discriminated_religion_to_kill
				}
			}
			
			#Reaction by exterminated discrimated pops
			add_radicals = {
				value = small_radicals
				culture = scope:discrimated_culture_to_kill
			}
			add_radicals = {
				value = small_radicals
				religion = scope:discriminated_religion_to_kill
			}
			
			#Chance for international community to discover genocide
			random_list = {
				11 = { }
				1 = {
					if = {
						limit = { NOT = { has_variable = genocide_discovered } }
						set_variable = genocide_discovered
				
						#Reaction by international community
						every_country = {
							limit = { 
								NOT = { THIS = ROOT }
								is_sufficiently_independent = yes
							}
							trigger_event = { id = genocide_system.1 }
						}
						
						#Notice to THIS that genocide has been discovered
						trigger_event = { id = genocide_system.2 }
						
					}
				}
			}
			
		}
		
	}

	complete = {
	
		#There are no longer any discrimated pops in THIS
		custom_tooltip = {
			text = je_genocide_system.tt
			NOT = { any_scope_pop = { pop_is_discriminated = yes } }
		}
		
	}
	
	on_complete = {
		
		#Reset genocide discovery
		remove_variable = genocide_discovered
		remove_variable = genocide_discovered_reacted
		
	}

	weight = 100

	should_be_pinned_by_default = yes

	transferable = no

	progressbar = no
}
