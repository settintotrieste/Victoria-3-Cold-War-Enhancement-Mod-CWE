dp_revolution = {
	requires_interest_marker = no
	enable_switch_sides = yes
	allow_negotiated_peace = no

	target_can_add_war_goals = no
	add_infamy_for_starting_initiator_wargoals = no
	
	war_goal = annex_country
	
	texture = "gfx/interface/icons/war_goals/annex_country.dds"
	
	selectable_in_lens = {
		always = no
	}	
	
	possible = {
		scope:initiator = {
			is_revolutionary = yes
		}
		scope:initiator.civil_war_origin_country = scope:target_country
	}

	additional_involvement_trigger = {
		NOT = { is_country_type = decentralized }
		aggressive_diplomatic_plays_permitted = yes
		
		#Potential participant is willing to get involved in other's affairs
		OR = { 
			has_interventionist_fp = yes
			scope:target_country = { is_subject_of = ROOT }
			scope:initiator = { is_subject_of = ROOT }
		}
	
		#Has an interest in the area
		scope:target_country = {
			capital = {
				region = {
					save_temporary_scope_as = target_capital_region
				}
			}
		}

		scope:initiator = {
			capital = {
				region = {
					save_temporary_scope_as = initiator_capital_region
				}
			}
		}
		
		OR = {
			has_interest_marker_in_region = scope:target_capital_region
			has_interest_marker_in_region = scope:initiator_capital_region
		}
	}
	
	on_weekly_pulse = {}

	on_war_begins = {}

	on_war_end = {}
}

	war_goal = annex_country
	
	texture = "gfx/interface/icons/war_goals/annex_country.dds"
	
	#selectable_in_lens = {
	#	always = no
	#}
	
	possible = {
	
		#THIS is not decentralized
		NOT = { is_country_type = decentralized }
		
		#THIS and Target are not revolutionary
		is_revolutionary = no is_secessionist = no
		scope:target_country = { is_revolutionary = no is_secessionist = no }
		
		#THIS has a higher rank than TARGET
		OR = { 
			country_rank > scope:target_country.country_rank
			scope:target_country = { 
				any_scope_state = { 
					is_homeland_of_country_cultures = ROOT
					is_homeland_of_country_cultures = scope:target_country 
				} 
			}
		}
		
		#THIS does not have a subject relationship with TARGET
		NOT = { is_subject_of = scope:target_country }
		scope:target_country = { NOT = { is_subject_of = root } }
		
		#THIS can launch aggressive wars
		aggressive_diplomatic_plays_permitted = yes
		
		#Bad relations with TARGET
		custom_tooltip = {
			text = DIPLO_ACTION_RELATIONS_POOR_OR_LESS

		}

			has_law = law_type:law_no_war

		}
		
		#THIS has some army
		NOT = { has_law = law_type:law_peasant_levies }
		
		#Ruler is jingoistic
		exists = ruler
		ruler = {
			OR = {
				has_ideology = ideology:ideology_autocratic
				has_ideology = ideology:ideology_power_hungry
			}
		}
		
		#No non-aggression_pact with TARGET
		has_no_non_aggression_pact_with_target = yes
